    if (num>1) right = num;
    else if (num < 1) right = 1;
    else {
        printf("root = 1");
        return 0;
    }
    root = (left + right)/2;
    while((root*root-num>1e-6)||(num - root * root > 1e-6)){
        if(root*root>num+1e-6){
            right = root;
        }
        else if (num>root*root+1e-6){
            left = root;

    }
